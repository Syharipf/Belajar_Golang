program tak

kamus
nilai : real

algoritma
input(nilai)
    if nilai < 2.0 do
		output("Poor")
	else if nilai >= 2.0 && nilai <= 2.75 do
	    output("Fair")
	else if nilai >= 2.76 && nilai <= 3.00 do
		output("Satisfactory")
	else if nilai >= 3.01 && nilai <= 3.50 do
		output("Very Good")
	else if nilai > 3.50 do
		output("Excellent")
    endif

endprogram
